🕷️ 博客來暢銷書爬蟲系統

這是一個全自動的博客來暢銷書排行榜爬蟲程式，能夠自動抓取書籍資訊包括排名、書名、作者、出版社、價格等詳細資料，並自動翻頁抓取多頁內容。程式具備完善的錯誤處理機制，能夠穩定運行並輸出結構化的資料檔案。 
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.by import By
import time
from webdriver_manager.chrome import ChromeDriverManager
from selenium.common.exceptions import NoSuchElementException, ElementNotInteractableException
import pandas as pd

def extract_books_data():
    """提取書籍資料的函數"""
    time.sleep(2)
    book_items = driver.find_elements(By.CSS_SELECTOR, ".item")
    
    for item in book_items:
        try:
            # 書名和連結
            title_element = item.find_element(By.CSS_SELECTOR, "h4 a")
            title = title_element.text.strip()
            book_link = title_element.get_attribute("href")
            
            # 作者
            try:
                author_element = item.find_element(By.CSS_SELECTOR, "ul li:first-child a")
                author = author_element.text.strip()
            except NoSuchElementException:
                author = "未知作者"
            
            # 價格
            try:
                price_element = item.find_element(By.CSS_SELECTOR, ".price_a b")
                price = price_element.text.strip()
            except NoSuchElementException:
                price = "價格未顯示"

            }

# 設定瀏覽器
options = webdriver.ChromeOptions()
options.add_argument("--disable-blink-features=AutomationControlled")

driver = webdriver.Chrome(
    service=Service(ChromeDriverManager().install()),
    options=options
)

try:
    # 前往博客來暢銷書頁面
    driver.get("https://www.books.com.tw/web/sys_saletopb/books/")
    driver.maximize_window()
    time.sleep(3)
    
    books_data = []
    pages_to_scrape = 3
    current_page = 1
    
    while current_page <= pages_to_scrape:
        print(f"正在爬取第 {current_page} 頁...")
        extract_books_data()
        
        if current_page < pages_to_scrape:
            try:
                next_button = driver.find_element(By.CSS_SELECTOR, "a[aria-label='前往下一頁']")
                if next_button.is_enabled():
                    driver.execute_script("arguments[0].click();", next_button)
                    time.sleep(3)
                    current_page += 1
                else:
                    break
            except NoSuchElementException:
                break
        else:
            break
    
    print(f"總共爬取了 {len(books_data)} 本書")
    
finally:
    driver.quit()

# 儲存資料
if books_data:
    df = pd.DataFrame(books_data)
    df.to_csv("博客來暢銷書排行榜.csv", index=False, encoding='utf-8-sig')
    df.to_excel("博客來暢銷書排行榜.xlsx", index=False)
    print("檔案儲存完成！")
